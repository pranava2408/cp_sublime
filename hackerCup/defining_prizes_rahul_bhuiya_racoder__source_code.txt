void solve(){
    ll n,m; cin>>n>>m;
    vector<ll> a(n),b(m);
    read(a); read(b);
    sort(all(b));
    vector<ll> pref(m);
    pref[0]=b[0];
    fr(i,1,m) pref[i]=pref[i-1]+b[i];
    // ll sum=accumulate(all(b),0ll);
    vector<pll> cnt;
    map<int,int> mpp;
    for(auto e:a) mpp[e]++;
    for(auto it:mpp) cnt.pb({it.ff,it.ss});
    ll net=0;
    ll c=1;
    for(auto [a,d]:cnt){
        net+=d*c;
        c++;
    }
    ll left=n;
    ll mx=cnt.size();
    ll unique=0;
    for(auto e:b) if(e) unique++;
    for(auto [a,c]:cnt){
        ll idx=upper_bound(all(b),left)-b.begin()-1;
        ll sum=0;
        if(idx>=0) sum=pref[idx];
        sum+=1ll*(m-1-idx)*left;
        debug(sum)
        if(net<=sum && mx<=unique){
            cout<<left;
            return;
        }
        mx--;
        net-=left;
        left-=c;
    }
    cout<<0;
}
