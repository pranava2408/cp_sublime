void solve()
{
    int n,m;
    cin>>n>>m;
    vector<int>a(n),b(m);
    get(a);
    get(b);
    sort(a.begin(),a.end());
    vector<int>cnt;
    for(int i=0;i<n;){
        int j=i;
        while(j<n && a[j]==a[i]) 
        ++j;
        cnt.push_back(j-i);
        i=j;
    }
    int k=(int)cnt.size();
    int maxl=min(k,m);
    sort(b.begin(),b.end());
    vector<int>pref(m+1,0);
    for(int i=0;i<m;++i) 
    pref[i+1]=pref[i]+b[i];
    auto summinb=[&](int K)->int{
        if(K<=0) 
        return 0;
        int p=upper_bound(b.begin(),b.end(),K)-b.begin();
        int s_le=pref[p];
        int cnt_gt=m-p;
        return s_le+cnt_gt*(int)K;
    };
    auto feasible=[&](int L)->bool{
        if(L==0) 
        return true;
        if(L>m) 
        return false;
        vector<int>freq(L+1,0);
        for(int t=1;t<=L;++t) 
        freq[t]=cnt[k-L+(t-1)];
        vector<int>cum(L+2,0),sumd(L+2,0);
        for(int v=L;v>=1;--v){
            cum[v]=cum[v+1]+freq[v];
            sumd[v]=sumd[v+1]+(int)v*freq[v];
        }
        int totald=sumd[1];
        int totals=pref[m];
        if(totald>totals) 
        return false;
        for(int v=L;v>=1;--v){
            int K=cum[v];
            if(K==0) continue;
            int left=sumd[v];
            int right=summinb(K);
            if(left>right) return false;
        }
        return true;
    };
    int lo=0,hi=maxl,best=0;
    while(lo<=hi){
        int mid=(lo+hi)>>1;
        if(feasible(mid))
        {
            best=mid;lo=mid+1;
        }
        else
        hi=mid-1;
    }
    int ans=0;
    for(int i=k-best;i<k;++i){
        if(i>=0) 
        ans+=cnt[i];
    }
    cout<<ans<<endl;
}

        cout<<"Case #"<<i<<": ";
