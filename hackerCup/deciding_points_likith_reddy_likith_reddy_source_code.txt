ll power(ll curr, ll zero, ll p = MOD)
{
    ll res = 1;
    curr = curr % p;
    while (zero > 0)
    {
        if (zero & 1)
            res = (res * curr) % p;
        zero = zero >> 1;
        curr = (curr * curr) % p;
    }
    return res;
}
