#include <bits/stdc++.h>
using namespace std;
int main()
{
    freopen("input1.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    long long num=0;
    long long t;
    cin>>t;
    while(t--)
    {
        num++;
        long long n,m;
        cin>>n>>m;
        vector <long long> a(n);
        for(long long i=0;i<n;i++) cin>>a[i];
        vector <long long> b(m);
        for(long long i=0;i<m;i++) cin>>b[i];
        map <long long,long long> mpp;
        for(long long i=0;i<n;i++)
        {
            mpp[a[i]]++;
        }
        long long sum=0;
        for(long long i=0;i<m;i++) sum=sum+b[i];
        long long ans=0,curr=0,unit=0,i=0;
        for(auto it=mpp.rbegin();it!=mpp.rend();it++)
        {
            i++;
            long long c=curr+it->second;
            long long d=unit+c;
            if(d>sum) break;
            ans=c;
            curr=c;
            unit=d;
            if(i>=m) break;
        }
        cout<<"Case #"<<num<<": "<<ans<<"\n";
    }
    return 0;
}
