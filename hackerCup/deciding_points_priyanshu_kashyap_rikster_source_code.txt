int isPrime(ll n){
  if(n<2) return 0;
  if(n%2==0) return 0;
  for(int i=2; i*i<=n; i++){
    if(n%i == 0) return 0;
  }
  return 1;
}

ll powerMod(ll a, ll b, ll m) {
    int ans=1;
    while(b)
    {
        if(b&1)
            ans=(ans*a)%m;
        b/=2;
        a=(a*a)%m;
    }
    return ans;
}

int log(ll a, ll b)
{
    int c = 0;
    while (a >= b)
    {
        c++;
        a /= b;
    }
    return c;
}


int powerOfTwo(int n) {
    int result = 1;
    int base = 2;
    while (n > 0) {
        if (n & 1LL) {
            result = (result * base) % MOD;
        }
        base = (base * base) % MOD;
        n >>= 1;
    }
    return result;
}
